WITH numbered AS (
    SELECT 
        shift_id,
        ROW_NUMBER() OVER (ORDER BY shift_id) AS rn
    FROM Shift
)
UPDATE Shift s
SET shift_date = DATE '2024-01-01'
                 + (((n.rn - 1) / 3)::int)
FROM numbered n
WHERE s.shift_id = n.shift_id;
